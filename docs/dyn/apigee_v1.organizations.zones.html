Returns:

    "data": { # SAML certificate and metadata. # Details of the certificate.
      "certificate": "A String", # SAML certificate.
      "issuer": "A String", # Issuer of the SAML certificate.
      "validTo": "A String", # Time the certificate expires in milliseconds since epoch.
      "valid": True or False, # Boolean flag that specifies whether the certificate is valid.
      "fingerprint": { # Fingerprint details of a SAML certificate. # Fingerprint of this SAML certificate.
